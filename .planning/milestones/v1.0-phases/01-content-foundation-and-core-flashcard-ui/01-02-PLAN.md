3. Create `src/i18n/utils.ts` -- helper functions:
   ```typescript
   import { ui, defaultLocale, type Locale, locales } from './ui';

   export function getLangFromUrl(url: URL): Locale {
     const [, lang] = url.pathname.split('/');
     if (locales.includes(lang as Locale)) return lang as Locale;
     return defaultLocale;
   }

   export function t(locale: Locale, key: keyof typeof ui[typeof defaultLocale]): string {
     return ui[locale][key] || ui[defaultLocale][key];
   }

   export function localePath(locale: Locale, path: string = ''): string {
     const base = import.meta.env.BASE_URL.replace(/\/$/, '');
     return `${base}/${locale}${path ? '/' + path : ''}`;
   }

   export { type Locale, defaultLocale, locales } from './ui';
   ```

5. Create `src/pages/[locale]/index.astro` -- the home page:
   ```astro
   ---
   import { getCollection } from 'astro:content';
   import BaseLayout from '../../layouts/BaseLayout.astro';
   import Header from '../../components/Header.astro';
   import SectionCard from '../../components/SectionCard.astro';
   import { t, type Locale, locales } from '../../i18n/utils';
   import { getAllSectionSlugs } from '../../i18n/sections';

   export function getStaticPaths() {
     return locales.map(locale => ({ params: { locale } }));
   }

   const { locale } = Astro.params as { locale: Locale };
   const allQuestions = await getCollection('questions');
   const sectionSlugs = getAllSectionSlugs();

   const sectionData = sectionSlugs.map(slug => ({
     slug,
     count: allQuestions.filter(q => q.data.section === slug).length,
   }));
   ---
   <BaseLayout title={t(locale, 'site.title')} description={t(locale, 'site.description')} locale={locale}>
     <Header locale={locale} />
     <main class="mx-auto max-w-4xl px-4 py-8 sm:px-6 sm:py-12">
       <div class="mb-8">
         <h1 class="text-3xl font-bold text-gray-900 sm:text-4xl">{t(locale, 'home.title')}</h1>
         <p class="mt-2 text-gray-600">{t(locale, 'home.subtitle')}</p>
       </div>
       <div class="grid gap-4 sm:grid-cols-2 lg:grid-cols-3">
         {sectionData.map(({ slug, count }) => (
           <SectionCard slug={slug} count={count} locale={locale} />
         ))}
       </div>
     </main>
   </BaseLayout>
   ```
   Grid: 1 column on mobile (< 640px), 2 columns on sm, 3 columns on lg. `max-w-4xl` constrains content width. `px-4` on mobile ensures no edge-to-edge content. Font sizes scale: 3xl mobile, 4xl desktop.
