1. Add the TYPE_ORDER constant and grouping logic in the frontmatter script section, after the existing `sectionQuestions` sort. Replace the current `questions` mapping with grouped logic:

```typescript
const TYPE_ORDER = ['basic', 'deep', 'trick', 'practical'] as const;

const questionKey = `${locale}_question` as const;
const answerKey = `${locale}_answer` as const;

// Group questions by type
const grouped = TYPE_ORDER
  .map(type => ({
    type,
    questions: sectionQuestions
      .filter(q => (q.data.type || 'basic') === type)
      .map(q => {
        const slug = q.id.split('/').pop() || q.id;
        return {
          slug,
          question: (q.data as any)[questionKey] as string,
          answer: (q.data as any)[answerKey] as string,
          type: (q.data.type || 'basic') as 'basic' | 'deep' | 'trick' | 'practical',
        };
      }),
  }))
  .filter(g => g.questions.length > 0);

// Total count for header
const totalCount = sectionQuestions.length;

// Type counts for summary (only non-zero types)
const typeCounts = TYPE_ORDER
  .map(type => ({
    type,
    count: sectionQuestions.filter(q => (q.data.type || 'basic') === type).length,
  }))
  .filter(tc => tc.count > 0);
```

Remove the old `const questions = sectionQuestions.map(...)` block and the old `questionKey`/`answerKey` declarations (they are now inside the new grouped logic above).
