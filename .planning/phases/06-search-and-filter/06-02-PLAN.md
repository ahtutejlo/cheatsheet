4. Add the tag filter `<script>` block. Insert it inside the `<BaseLayout>` block, after the existing anchor link handler `</script>` and before `</BaseLayout>`. Add this script:
   ```html
   <script>
     function initTagFilter() {
       const filterBar = document.getElementById('filter-bar');
       const filterTagName = document.getElementById('filter-tag-name');
       const clearBtn = document.getElementById('clear-filter');
       let activeTag: string | null = null;

       function applyFilter(tag: string | null) {
         activeTag = tag;

         // Update filter bar visibility
         if (filterBar) {
           filterBar.classList.toggle('hidden', !tag);
           // When visible, ensure flex display
           if (tag) filterBar.classList.add('flex');
           else filterBar.classList.remove('flex');
         }
         if (filterTagName && tag) {
           filterTagName.textContent = tag;
         }

         // Update tag badge active states
         document.querySelectorAll('.tag-filter').forEach(btn => {
           const btnTag = (btn as HTMLElement).dataset.tag;
           const isActive = btnTag === tag;
           btn.classList.toggle('bg-blue-500', isActive);
           btn.classList.toggle('text-white', isActive);
           btn.classList.toggle('dark:bg-blue-600', isActive);
           btn.classList.toggle('dark:text-white', isActive);
           // Remove default classes when active
           btn.classList.toggle('bg-gray-100', !isActive);
           btn.classList.toggle('text-gray-600', !isActive);
           btn.classList.toggle('dark:bg-gray-700', !isActive);
           btn.classList.toggle('dark:text-gray-300', !isActive);
         });

         // Show/hide flashcards
         document.querySelectorAll('.flashcard').forEach(card => {
           if (!tag) {
             (card as HTMLElement).style.display = '';
             return;
           }
           const cardTags = (card as HTMLElement).dataset.tags?.split(',') || [];
           (card as HTMLElement).style.display = cardTags.includes(tag) ? '' : 'none';
         });

         // Show/hide empty type groups
         document.querySelectorAll('[data-type-group]').forEach(group => {
           if (!tag) {
             (group as HTMLElement).style.display = '';
             return;
           }
           const visibleCards = group.querySelectorAll('.flashcard:not([style*="display: none"])');
           (group as HTMLElement).style.display = visibleCards.length ? '' : 'none';
         });
       }

       // Tag badge click handler (event delegation on main)
       document.querySelector('main')?.addEventListener('click', (e) => {
         const btn = (e.target as HTMLElement).closest('.tag-filter');
         if (!btn) return;
         e.preventDefault();
         // Prevent details from toggling when clicking tag inside summary
         e.stopPropagation();
         const tag = (btn as HTMLElement).dataset.tag || null;
         applyFilter(tag === activeTag ? null : tag);
       });

       // Clear filter button
       clearBtn?.addEventListener('click', () => applyFilter(null));
     }

     initTagFilter();
   </script>
   ```
